34void FauxmoESP::processRequest(const String& url, const String& body, const String& method, String& response, uint16_t* resultCode)
35{
36 // GET /api/<username>/lights
37 if (method == "GET" && url.startsWith("/api/") && url.endsWith("/lights"))
38 {
39 onLightList(response, resultCode);
40 }
41
42 // PUT /api/<username>/lights/<lightId>/state
43 else if (method == "PUT" && url.startsWith("/api/") && url.endsWith("/state"))
44 {
45 String username = url.substring(5, url.indexOf("/lights/"));
46 String lightId = url.substring(url.indexOf("/lights/") + 8, url.indexOf("/state"));
47
48 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /api/%s/lights/%s/state request!\n", username.c_str(),
49 lightId.c_str())
50
51 onLightStateChange(lightId.toInt(), body, response, resultCode);
52 }
53
54 // GET /api/<username>/lights/<lightId>
55 else if (method == "GET" && url.startsWith("/api/") && url.endsWith("/state"))
56 {
57 String username = url.substring(5, url.indexOf("/lights/"));
58 String lightId = url.substring(url.indexOf("/lights/") + 8, url.indexOf("/state"));
59
60 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /api/%s/lights/%s/state request!\n", username.c_str(),
61 lightId.c_str())
62
63 onLightStateRequest(lightId.toInt(), response, resultCode);
64 }
65 // GET /description.xml
66 else if (method == "GET" && url == "/description.xml")
67 {
68 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /description.xml request!\n")
69 #if defined(ESP8266)
70 #elif defined(ESP32)
71 response = SSDP.getSchema();
72 #endif
73 *resultCode = 200;
74 }
75 // POST /api
76 else if (method == "POST" && url == "/api")
77 {
78 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /api request!\n")
79 response = R"({"success":{"username":")" + this->_username + R"("}})";
80 *resultCode = 200;
81 }
82 else
83 {
84 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received unknown request!\n")
85 *resultCode = 404;
86 }
87}

90{
91 String mac = WiFi.macAddress();

94

103 SSDP.setModelName("Philips hue bridge 2012");
104 SSDP.setModelNumber("929000226503");
105 SSDP.setModelURL("http://www.meethue.com");

122void FauxmoESP::setupWebServer()
123{
124 #if defined(ESP8266)
125 this->webServer = new ESP8266WebServer(_webServerPort);
126 #elif defined(ESP32)
127 this->webServer = new WebServer(_webServerPort);
128 #endif
129
130 // Allows the user to turn the light on and off, modify the hue and effects.
131 this->webServer->on(UriRegex("/api/([A-Za-z0-9]+)/lights/([0-9]+)/state"), HTTP_PUT, [this]
132 {
133 String username = webServer->pathArg(0);
134 String lightId = webServer->pathArg(1);
135 if (!checkUsername(username))
136 {
137 return;
138 }
139
140 if (!webServer->hasArg("plain"))
141 {
142 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "No body\n")
143 webServer->send(400, "application/json", R"({"error":{"type":3,"address":"/lights/)" + String(lightId) + R"(","description":"resource, /lights/)" +
144 String(lightId) + ", not available\"}}");
145 return;
146 }
147 String body = webServer->arg("plain");
148
149 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /api/%s/lights/%s/state request!\n", username.c_str(),
150 lightId.c_str())
151
152 String response;
153 uint16_t resultCode = 0;
154 onLightStateChange(lightId.toInt(), body, response, &resultCode);
155
156 webServer->send(resultCode, "application/json", response);
157 });
158
159 // Gets the attributes and state of a given light.
160 this->webServer->on(UriRegex("/api/([A-Za-z0-9]+)/lights/([0-9]+)"), HTTP_GET, [this]
161 {
162 String username = webServer->pathArg(0);
163 String lightId = webServer->pathArg(1);
164 if (!checkUsername(username))
165 {
166 return;
167 }
168
169 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /api/%s/lights/%s request!\n", username.c_str(), lightId.c_str())
170 String response;
171 uint16_t resultCode = 0;
172 onLightStateRequest(lightId.toInt(), response, &resultCode);
173
174 webServer->send(resultCode, "application/json", response);
175 });
176
177 // Gets a list of all lights that have been discovered by the bridge.
178 this->webServer->on(UriRegex("/api/([A-Za-z0-9]+)/lights"), HTTP_GET, [this]
179 {
180 String username = webServer->pathArg(0);
181 if (!checkUsername(username))
182 {
183 return;
184 }
185
186 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /api/%s/lights request!\n", username.c_str())
187 String response;
188 uint16_t resultCode = 0;
189 onLightList(response, &resultCode);
190
191 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Sending light list!\n")
192 webServer->send(resultCode, "application/json", response);
193 });
194
195 this->webServer->on("/description.xml", HTTP_GET, [this]
196 {
197 #if defined(ESP8266)
198 SSDP.schema(webServer->client());
199 #elif defined(ESP32)
200 webServer->send(200, "text/xml", SSDP.getSchema());
201 #endif
202 });
203
204 this->webServer->on("/api", HTTP_POST, [this]
205 {
206 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received /api POST request!\n")
207 webServer->send(200, "application/json", R"([{"success":{"username":")" + _username + "\"}}]");
208 });
209
210 this->webServer->onNotFound([this]
211 {
212 auto method = "GET";
213 if (webServer->method() == HTTP_POST)
214 {
215 method = "POST";
216 }
217 else if (webServer->method() == HTTP_PUT)
218 {
219 method = "PUT";
220 }
221 else if (webServer->method() == HTTP_DELETE)
222 {
223 method = "DELETE";
224 }
225 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received unknown request! %s %s\n", method, webServer->uri().c_str())
226 webServer->send(404, "text/plain", "Not found");
227 });
228
229 this->webServer->begin();
230}

292void FauxmoESP::onLightStateChange(uint16_t lightId, const String& body, String& response, uint16_t* resultCode)
293{
294 auto* light = getLightByDeviceId(lightId);
295 if (light == nullptr)
296 {
297 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Light with id %d not found\n", lightId)
298 response = R"({"error":{"type":3,"address":"/lights/)" + String(lightId) + R"(","description":"resource, /lights/)" +
299 String(lightId) + ", not available\"}}";
300 *resultCode = 404;
301 return;
302 }
303
304 auto stateChange = LightStateChange(body);
305 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Received state change: %s (%s)\n", stateChange.getChanges(lightId).c_str(),
306 body.c_str())
307
308 // Modifying hue, saturation, brightness, effect, ct or xy when the light is off will return 201 error.
309 if (!light->state.isOn && (!stateChange.isOnSet() || !stateChange.getIsOn()))
310 {
311 DEBUG_MSG_FAUXMO(
312 FAUXMO_LOG_TAG "Modifying hue, saturation, brightness, effect, ct or xy when the light is off is not allowed\n")
313 response = R"({"error":{"type": 201, "address": "/lights/)" + String(lightId)
314 + R"(/state", "description": "Modifying hue, saturation, brightness, effect, ct or xy when the light is off is not allowed."}})";
315 *resultCode = 201;
316 return;
317 }
318
319 if (light->state.isLightReachable)
320 {
321 if (_onStateChange != nullptr)
322 {
323 this->_onStateChange(light, &stateChange);
324 }
325
326 const String& changesMade = stateChange.getChanges(lightId);
327 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Light %s State changed: %s!\n", light->name.c_str(), changesMade.c_str())
328 response = changesMade;
329 *resultCode = 200;
330 return;
331 }
332 else
333 {
334 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "State is not valid for light %s: light is unreachable!\n", light->name.c_str())
335 response = R"({"error":{"type": 201, "address": "/lights/)" + String(lightId)
336 + R"(/state", "description": "State is not valid."}})";
337 *resultCode = 400;
338 return;
339 }
340
341 assert(false && "Should not be possible to reach this point");
342}
