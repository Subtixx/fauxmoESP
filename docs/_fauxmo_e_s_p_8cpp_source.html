47void FauxmoESP::_sendUDPResponse()
48{
49 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Responding to M-SEARCH request\n")
50
51 IPAddress ip = WiFi.localIP();
52 String mac = WiFi.macAddress();
53 mac.replace(":", "");
54 mac.toLowerCase();
55
56 char response[strlen(FAUXMO_UDP_RESPONSE_TEMPLATE) + 128];
57 snprintf_P(
58 response, sizeof(response),
59 FAUXMO_UDP_RESPONSE_TEMPLATE,
60 ip[0], ip[1], ip[2], ip[3],
61 _tcp_port,
62 mac.c_str(), mac.c_str()
63 );
64
65 #if DEBUG_FAUXMO_VERBOSE_UDP
66 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "UDP response sent to %s:%d\n%s", _udp.remoteIP().toString().c_str(), _udp.remotePort(), response);
67 #endif
68
69 _udp.beginPacket(_udp.remoteIP(), _udp.remotePort());
70 #if defined(ESP32)
71 _udp.printf(response);
72 #else
73 _udp.write(response);
74 #endif
75 _udp.endPacket();
76
77}

79void FauxmoESP::_handleUDP()
80{
81 int len = _udp.parsePacket();
82 if (len > 0)
83 {
84 unsigned char data[len + 1];
85 _udp.read(data, len);
86 data[len] = 0;
87
88 #if DEBUG_FAUXMO_VERBOSE_UDP
89 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "UDP packet received\n%s", (const char *) data);
90 #endif
91
92 String request = String((const char*)data);
93 if (request.indexOf("M-SEARCH") >= 0)
94 {
95 if ((request.indexOf("ssdp:discover") > 0) || (request.indexOf("upnp:rootdevice") > 0)
96 || (request.indexOf("device:basic:1") > 0))
97 {
98 _sendUDPResponse();
99 }
100 }
101 }
102}

115void FauxmoESP::_sendTCPResponse(AsyncClient* client, const char* code, const char* body, const char* mime)
116{
117 char headers[strlen_P(FAUXMO_TCP_HEADERS) + 32];
118 snprintf_P(
119 headers, sizeof(headers),
120 FAUXMO_TCP_HEADERS,
121 code, mime, strlen(body)
122 );
123
124 #if DEBUG_FAUXMO_VERBOSE_TCP
125 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Response:\n%s%s\n", headers, body);
126 #endif
127
128 client->write(headers);
129 client->write(body);
130}

147
148 if (all)

150 snprintf_P(
151 buffer, sizeof(buffer),
152 FAUXMO_DEVICE_JSON_TEMPLATE,
153 device.name.c_str(),
154 device.uniqueid,
155 device.state ? "true" : "false",
156 device.value,
157 device.hue,
158 device.sat
159 );
160 }
161 else
162 {
163 snprintf_P(
164 buffer, sizeof(buffer),
165 FAUXMO_DEVICE_JSON_TEMPLATE_SHORT,
166 device.name.c_str(),
167 device.uniqueid
168 );
169 }
170

211bool FauxmoESP::_onTCPList(AsyncClient* client, const String& url, [[maybe_unused]] const String& body)
212{
213 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Handling list request\n")
214
215 // Get the index
216 int pos = url.indexOf("lights");
217 if (-1 == pos)
218 {
219 return false;
220 }
221
222 // Get the id
223 unsigned char id = url.substring(pos + 7).toInt();
224
225 // This will hold the response string
226 String response;
227
228 // Client is requesting all devices
229 if (0 == id)
230 {
231 response += "{";
232 for (size_t i = 0; i < _devices.size(); i++)
233 {
234 if (i > 0) response += ",";
235 response += "\"" + String(i + 1) + "\":" + _deviceJson(i, false); // send short template
236 }
237 response += "}";
238
239 // Client is requesting a single device
240 }
241 else
242 {
243 response = _deviceJson(id - 1);
244 }
245
246 char responseStr[response.length() + 1];
247 response.toCharArray(responseStr, response.length() + 1);
248
249 _sendTCPResponse(client, "200 OK", responseStr, "application/json");
250
251 return true;
252

262bool FauxmoESP::_onTCPControl(AsyncClient* client, const String& url, const String& body)
263{
264 // "devicetype" request
265 if (body.indexOf("devicetype") > 0)
266 {
267 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Handling devicetype request\n")
268
269 _sendTCPResponse(client, "200 OK", R"([{"success":{"username": "2WLEDHardQrI3WHYTHoMcXHgEspsM8ZZRpSKtBQr"}}])",
270 "application/json");
271 return true;
272 }
273
274 // "state" request
275 if ((url.indexOf("state") > 0) && (body.length() > 0))
276 {
277 // Get the index
278 int pos = url.indexOf("lights");
279 if (-1 == pos) return false;
280
281 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Handling state request\n")
282
283 // Get the index
284 unsigned char id = url.substring(pos + 7).toInt();
285 if (id > 0)
286 {
287 --id;
288 // Brightness
289 pos = body.indexOf("bri");
290 if (pos > 0)
291 {
292 unsigned char value = body.substring(pos + 5).toInt();
293 _devices[id].value = value;
294 _devices[id].state = (value > 0);
295 }
296 else if (body.indexOf("false") > 0)
297 {
298 _devices[id].state = false;
299 }
300 else
301 {
302 _devices[id].state = true;
303 if (0 == _devices[id].value) _devices[id].value = 255;
304 }
305
306 // Hue
307 pos = body.indexOf("hue");
308 if (pos > 0)
309 {
310 unsigned char hue = body.substring(pos + 5).toInt();
311 _devices[id].hue = hue;
312 }
313
314 // Saturation
315 pos = body.indexOf("sat");
316 if (pos > 0)
317 {
318 unsigned char sat = body.substring(pos + 5).toInt();
319 _devices[id].sat = sat;
320 }
321
322 char response[strlen_P(FAUXMO_TCP_STATE_RESPONSE) + 10];
323 snprintf_P(
324 response, sizeof(response),
325 FAUXMO_TCP_STATE_RESPONSE,
326 id + 1, _devices[id].state ? "true" : "false", id + 1,
327 _devices[id].value, id + 1, _devices[id].hue, id + 1, _devices[id].sat
328 );
329 _sendTCPResponse(client, "200 OK", response, "text/xml");
330
331 if (_setCallback)
332 {
333 _setCallback(id, &_devices[id]);
334 }
335 return true;
336 }
337 }
338 return false;
339}

349bool FauxmoESP::_onTCPRequest(AsyncClient* client, bool isGet, const String& url, const String& body)
350{
351 if (!_fauxmoEnabled)
352 {
353 return false;
354 }
355
356 #if DEBUG_FAUXMO_VERBOSE_TCP
357 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "isGet: %s\n", isGet ? "true" : "false");
358 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "URL: %s\n", url.c_str());
359 if (!isGet)
360 {
361 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Body:\n%s\n", body.c_str());
362 }
363 #endif
364
365 if (url.equals("/description.xml"))
366 {
367 return _onTCPDescription(client, url, body);
368 }
369
370 if (url.startsWith("/api"))
371 {
372 if (isGet)
373 {
374 return _onTCPList(client, url, body);
375 }
376 else
377 {
378 return _onTCPControl(client, url, body);
379 }
380 }
381
382 if (_redirect_port != 0)
383 {
384 char response[strlen_P(FAUXMO_REDIRECT) + 5];
385 snprintf_P(
386 response, sizeof(response),
387 FAUXMO_REDIRECT,
388 _redirect_port
389 );
390 _sendTCPResponse(client, "200 OK", response, "text/html");
391 }
392
393 return false;
394}

396bool FauxmoESP::_onTCPData(AsyncClient* client, void* data, size_t len)
397{
398 if (!_fauxmoEnabled)
399 {
400 return false;
401 }
402
403 char* p = static_cast<char*>(data);
404 p[len] = 0;
405
406 #if DEBUG_FAUXMO_VERBOSE_TCP
407 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "TCP request\n%s\n", p);
408 #endif
409
410 // Method is the first word of the request
411 char* method = p;
412
413 while (*p != ' ') p++;
414 *p = 0;
415 p++;
416
417 // Split word and flag start of url
418 char* url = p;
419
420 // Find next space
421 while (*p != ' ') p++;
422 *p = 0;
423 p++;
424
425 // Find double line feed
426 unsigned char c = 0;
427 while ((*p != 0) && (c < 2))
428 {
429 if (*p != '\r')
430 {
431 c = (*p == '\n') ? c + 1 : 0;
432 }
433 p++;
434 }
435 char* body = p;
436
437 bool isGet = (strncmp(method, "GET", 3) == 0);
438
439 return _onTCPRequest(client, isGet, url, body);
440}

447void FauxmoESP::_onTCPClient(AsyncClient* client)
448{
449 if (_fauxmoEnabled)
450 {
451 [[maybe_unused]] unsigned int i = 0;
452 for (auto& _tcpClient : _tcpClients)
453 {
454 if (!_tcpClient || !_tcpClient->connected())
455 {
456 _tcpClient = client;
457 client->onAck([](void* s, AsyncClient* c, size_t len, uint32_t time)
458 {
459 }, nullptr);
460
461 client->onData([this](void* s, AsyncClient* c, void* data, size_t len)
462 {
463 _onTCPData(c, data, len);
464 }, nullptr);
465
466 client->onDisconnect([&_tcpClient](void* s, AsyncClient* c)
467 {
468 if (_tcpClient != nullptr)
469 {
470 _tcpClient->free();
471 _tcpClient = nullptr;
472 }
473 else
474 {
475 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Client %d already disconnected\n", i)
476 }
477 delete c;
478 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Client #%d disconnected\n", i)
479 }, nullptr);
480
481 client->onError([](void* s, AsyncClient* c, int8_t error)
482 {
483 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Error %s (%d) on client #%d\n", c->errorToString(error), error, i)
484 }, nullptr);
485
486 client->onTimeout([](void* s, AsyncClient* c, uint32_t time)
487 {
488 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Timeout on client #%d at %i\n", i, time)
489 c->close();
490 }, nullptr);
491
492 client->setRxTimeout(FAUXMO_RX_TIMEOUT);
493
494 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Client #%d connected\n", i)
495 return;
496 }
497 i++;
498 }
499
500 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Rejecting - Too many connections\n")
501 }
502 else
503 {
504 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Rejecting - Disabled\n")
505 }
506
507 client->onDisconnect([](void* s, AsyncClient* c)
508 {
509 c->free();
510 delete c;
511 });
512 client->close(true);
513}

520[[maybe_unused]] unsigned char FauxmoESP::addDevice(const char* deviceName)
521{
522 fauxmoesp_device_t device;
523 unsigned char deviceId = _devices.size();
524
525 // init properties
526 device.name = strdup(deviceName);
527 device.state = false;
528 device.value = 0;

534

539
540 DEBUG_MSG_FAUXMO(FAUXMO_LOG_TAG "Device '%s' added as #%d\n", deviceName, deviceId)
